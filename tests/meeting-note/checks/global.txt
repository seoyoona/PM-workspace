# /meeting-note offline harness global checks (Wave 1-B)
# TAB-separated. Comments start with #.
# global checks support must_match / must_not_match only (per harness-common.sh run_global_check).
# in-section / sub-heading checks live in caseNN.txt or are expressed as file-level patterns here.

# Frontmatter
must_match	^---$	frontmatter delimiter
must_match	^status: (Confirmed-1|Confirmed-2|Cancelled)$	status enum
must_match	^case: case[0-9]+$	case ID present
must_match	^client: 	client field
must_match	^project: 	project field
must_match	^meeting_date: [0-9]{4}-[0-9]{2}-[0-9]{2}$	meeting_date YYYY-MM-DD
must_match	^notion_comm_db: 	Notion comm DB ref
must_match	^gate_choice: 	gate_choice present (value verified per case)
must_match	^existing_match_count: [0-9]+$	existing_match_count integer
must_match	^action_items_count: [0-9]+$	action_items_count integer

# Required top-level sections
must_match	## 1\. Step 3 Confirm Gate	section 1
must_match	## 2\. PM Choice	section 2
must_match	## 3\. Part 1	section 3 — Part 1
must_match	## 4\. Part 2	section 4 — Part 2
must_match	## 5\. Part 3	section 5 — Part 3
must_match	## 7\. Notes	section 7

# Part 1 §3 sub-headings must be present at file level (### 유지 — bold 평탄화 X)
must_match	^### 미팅 정보$	§3 미팅 정보 sub-heading
must_match	^### 한눈에 보기$	§3 한눈에 보기 sub-heading
must_match	^### Action Items$	§3 Action Items sub-heading

# Part 2 의무 closing line (skill body L199)
must_match	Please review and let me know if anything needs more effort than expected\.	Part 2 closing line

# No auto-chain triggers (auto-execute /to-spec / etc forbidden)
must_not_match	^Running /(qa-plan|issue-ticket|to-spec|client-chat|dev-chat)\b	no auto-chain
must_not_match	→ 자동 실행	no auto-execution language

# No raw MCP write call traces (snapshot represents intent, not actual calls)
must_not_match	mcp__notion-cigro__notion-create-pages\b	no raw create-pages trace
must_not_match	mcp__notion-cigro__notion-update-page\b	no raw update-page trace

# No Mermaid diagrams (v1 baseline)
must_not_match	```mermaid	no Mermaid block
must_not_match	flowchart [TLBR]D	no Mermaid flowchart
must_not_match	graph [TLBR]D	no Mermaid graph

# No AI-invented effort estimates
must_not_match	[0-9]+ ?MD\b	no MD effort
must_not_match	[0-9]+ ?(시간|hours?|hrs?)\b	no hour effort

# No filler phrases (file-level — Action Items §3 has sub-headings preventing in-section extract)
must_not_match	(Please consider|It seems|There is a problem with)	no filler phrases
